路由跳转


    let routeTo = (url,type) => {
        switch(type){
            case 'nT':
            return wx.navigateTo({url});
            case 'rT':
            return wx.redirectTo({url});
            case 'rL':
            return wx.reLaunch({url});
            case 'sT':
            return wx.switchTab({url});
            default: wx.navigateBack({delta: 1})
            break
        }
    }
